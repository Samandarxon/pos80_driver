  if [ $i -lt 25 ]; then
    # 20-24 ‚Üí 201-205
    room_num=$((i - 19))
    room="20$room_num"
  elif [ $i -lt 30 ]; then

    # 30 ‚Üí 201

    -H "Content-Type: application/json" \

done
